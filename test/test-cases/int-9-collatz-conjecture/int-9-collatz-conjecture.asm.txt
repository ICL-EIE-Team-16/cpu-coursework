ADDIU $a0, $zero, 3                    #stores the x value
ADDU $v0, $zero, $zero                 #counter
ADDIU $a1, $zero, 6                   #stores START address
ADDIU $a2, $zero, 3                    #stores 3
ADDIU $t0, $zero, 1                    #stores 1
BEQ $a0, $t1, 0xC                     #if x=1 branch 12 forwards to end         START
ADDIU $v0, $v0, 1                      #increment counter by 1
ANDI $a3, $a0, 1
BNE $a3, $zero, 4
nop                   
SRL $a0, $a0, 1                       #even section, x=x/2                      EVEN
JR $a1                                #back to start
nop
MULT $a0, $a2                         #odd section x=3x+1                       ODD
MFLO $a0
ADDIU $a0, $a0, 1
JR $a1                                #back to start
nop
JR $zero                              #terminate
nop
